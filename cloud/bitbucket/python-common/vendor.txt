# Conditionally required by ipython on MacOS
appnope --platform macosx_10_6_intel

# Download binary version of grpcio-tools because it takes a lot of time to compile it, but it's used only for
# protos generation and is not uses on production.
grpcio-tools==1.15.0 --platform manylinux1_x86_64 --no-deps
grpcio-tools==1.15.0 --platform macosx_10_9_intel --no-deps

# To increase the speed of protocol decoding we must use binary version of protobuf package
protobuf --platform manylinux1_x86_64 --no-deps
protobuf --platform macosx_10_9_intel --no-deps

# Download binary version of gauthling packages because it's hard to compile them (depends on Rust) and
# because compilation takes a lot of time.
yc-auth-token --platform linux_x86_64 --no-deps
yc-auth-token --platform macosx_10_6_intel --no-deps
gauthling-daemon --platform linux_x86_64 --no-deps
gauthling-daemon --platform macosx_10_6_intel --no-deps
